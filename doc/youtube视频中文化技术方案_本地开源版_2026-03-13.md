### Step 1: 视频下载与音频提取

```python
# src/step1_download.py
import os
import subprocess
from pydub import AudioSegment

def download_video(youtube_url: str, output_dir: str = "temp") -> tuple:
    """下载视频并提取音频"""
    os.makedirs(output_dir, exist_ok=True)

    # 下载视频
    video_cmd = [
        "yt-dlp",
        "-f", "bestvideo[ext=mp4]+bestaudio[ext=m4a]/best[ext=mp4]/best",
        "-o", f"{output_dir}/video.%(ext)s",
        youtube_url
    ]
    subprocess.run(video_cmd, check=True)

    # 提取音频 (16kHz WAV)
    audio_cmd = [
        "ffmpeg",
        "-i", f"{output_dir}/video.mp4",
        "-ar", "16000",
        "-ac", "1",
        "-f", "wav",
        f"{output_dir}/audio.wav"
    ]
    subprocess.run(audio_cmd, check=True)

    return f"{output_dir}/video.mp4", f"{output_dir}/audio.wav"
```

### Step 2: 语音识别 (FunASR)

```python
# src/step2_transcribe.py
from funasr import AutoModel
import json

def transcribe_audio(audio_path: str) -> dict:
    """语音识别，返回带时间戳的文本"""

    # 加载模型 (首次运行会自动下载)
    model = AutoModel(
        model="paraformer-large-asr_nat-zh-cn-16k-common-vocab8404-pytorch",
        vad_model="fsmn-vad",
        punc_model="ct-punc",
        device="cuda",  # 使用 GPU
        disable_update=True  # 禁用更新检查
    )

    # 识别
    result = model.generate(
        input=audio_path,
        batch_size=16
    )

    # 解析结果
    sentences = []
    if isinstance(result, list) and len(result) > 0:
        output = result[0]

        if "sentence_info" in output:
            # 有详细时间戳
            for sent in output["sentence_info"]:
                sentences.append({
                    "text": sent.get("text", ""),
                    "start": sent.get("start", 0),
                    "end": sent.get("end", 0)
                })
        else:
            # 只有文本
            sentences.append({
                "text": output.get("text", ""),
                "start": 0,
                "end": None  # 后续计算
            })

    return {
        "sentences": sentences,
        "full_text": " ".join([s["text"] for s in sentences])
    }
```

### Step 3: 翻译 (Ollama + Qwen3)

```python
# src/step3_translate.py
import requests
import time

TRANSLATION_PROMPT = """你是一个专业的视频字幕翻译官。请将以下英文文本翻译成中文，要求：

1. 保持口语化和自然流畅
2. 保留原始句子的情感和语气
3. 控制句子长度，适合视频字幕显示
4. 专业术语保持准确

英文原文：
{english_text}

中文翻译："""

def translate_text(english_text: str, retries: int = 3) -> str:
    """调用 Ollama 翻译文本"""

    prompt = TRANSLATION_PROMPT.format(english_text=english_text)

    for attempt in range(retries):
        try:
            response = requests.post(
                "http://localhost:11434/api/generate",
                json={
                    "model": "qwen3:7b",
                    "prompt": prompt,
                    "stream": False,
                    "options": {
                        "temperature": 0.3,
                        "top_p": 0.9
                    }
                },
                timeout=120
            )

            if response.status_code == 200:
                result = response.json()
                translation = result.get("response", "").strip()

                # 清理输出 (移除可能的前缀)
                if translation.startswith("中文翻译："):
                    translation = translation.replace("中文翻译：", "").strip()

                return translation
            else:
                print(f"Ollama 返回错误: {response.status_code}")

        except requests.exceptions.RequestException as e:
            print(f"请求失败 (尝试 {attempt + 1}/{retries}): {e}")
            if attempt < retries - 1:
                time.sleep(2 ** attempt)  # 指数退避
            else:
                raise

    raise Exception("翻译失败，已达到最大重试次数")


def translate_sentences(sentences: list) -> list:
    """批量翻译句子"""
    translated = []

    for i, sentence in enumerate(sentences):
        print(f"翻译句子 {i+1}/{len(sentences)}...")
        chinese_text = translate_text(sentence["text"])

        translated.append({
            "english": sentence["text"],
            "chinese": chinese_text,
            "start": sentence["start"],
            "end": sentence["end"]
        })

    return translated
```

### Step 4: 语音合成 (CosyVoice)

```python
# src/step4_synthesize.py
import os
from cosyvoice.cli.cosyvoice import CosyVoice
import soundfile as sf
import numpy as np
from pydub import AudioSegment

def synthesize_speech(sentences: list, output_dir: str = "temp") -> str:
    """批量合成语音并拼接"""

    os.makedirs(output_dir, exist_ok=True)

    # 加载模型
    cosyvoice = CosyVoice('pretrained_models/CosyVoice-300M')

    # 获取可用发音人
    speakers = cosyvoice.list_avaliable_spks()
    print(f"可用发音人: {speakers}")

    # 选择发音人 (默认: 中文女)
    speaker = "中文女" if "中文女" in speakers else speakers[0]

    # 逐句合成
    audio_segments = []
    total_duration = 0

    for i, sentence in enumerate(sentences):
        print(f"合成句子 {i+1}/{len(sentences)}...")

        # 合成
        result = cosyvoice.inference_sft(
            text=sentence["chinese"],
            speaker=speaker
        )

        # 保存临时文件
        segment_path = f"{output_dir}/segment_{i:04d}.wav"
        audio_data = result['tts_speech'].numpy().flatten()
        sf.write(segment_path, audio_data, 22050)

        # 加载音频段
        segment = AudioSegment.from_wav(segment_path)
        audio_segments.append({
            "audio": segment,
            "start": sentence["start"],
            "end": sentence["end"],
            "duration": len(segment) / 1000.0  # 秒
        })

        total_duration += len(segment) / 1000.0

    # 拼接所有音频段
    final_audio = audio_segments[0]["audio"]
    for segment in audio_segments[1:]:
        final_audio += segment["audio"]

    # 保存最终音频
    output_path = f"{output_dir}/chinese_audio.wav"
    final_audio.export(output_path, format="wav")

    print(f"语音合成完成，总时长: {total_duration:.2f}秒")

    return output_path, audio_segments
```

### Step 5: 视频合成

```python
# src/step5_merge.py
import os
import ffmpeg
import subprocess

def generate_srt(sentences: list, output_path: str = "temp/subtitle.srt"):
    """生成 SRT 字幕文件"""

    def format_time(seconds: float) -> str:
        """格式化时间 (秒 -> HH:MM:SS,mmm)"""
        hrs = int(seconds // 3600)
        mins = int((seconds % 3600) // 60)
        secs = int(seconds % 60)
        ms = int((seconds % 1) * 1000)
        return f"{hrs:02d}:{mins:02d}:{secs:02d},{ms:03d}"

    srt_content = []
    for i, sentence in enumerate(sentences, 1):
        start = format_time(sentence["start"])
        end = format_time(sentence["end"])
        text = sentence["chinese"]

        srt_content.append(f"{i}")
        srt_content.append(f"{start} --> {end}")
        srt_content.append(text)
        srt_content.append("")

    os.makedirs(os.path.dirname(output_path), exist_ok=True)

    with open(output_path, "w", encoding="utf-8") as f:
        f.write("\n".join(srt_content))

    return output_path


def merge_video(
    video_path: str,
    audio_path: str,
    subtitle_path: str,
    output_path: str = "output/final_video.mp4"
):
    """合并视频、音频和字幕"""

    os.makedirs(os.path.dirname(output_path), exist_ok=True)

    # 使用 ffmpeg 合并
    cmd = [
        "ffmpeg",
        "-i", video_path,
        "-i", audio_path,
        "-vf", f"subtitles={subtitle_path}:force_style='Fontsize=24,PrimaryColour=&H00FFFFFF,OutlineColour=&H00000000,BackColour=&H80000000,BorderStyle=4,Outline=1,Shadow=0,Alignment=2'",
        "-map", "0:v:0",
        "-map", "1:a:0",
        "-c:v", "libx264",
        "-preset", "medium",
        "-crf", "23",
        "-c:a", "aac",
        "-b:a", "192k",
        "-shortest",
        "-y",
        output_path
    ]

    print(f"正在合成视频: {output_path}")
    result = subprocess.run(cmd, capture_output=True, text=True)

    if result.returncode != 0:
        print(f"FFmpeg 错误: {result.stderr}")
        raise Exception("视频合成失败")

    print(f"✅ 视频合成完成: {output_path}")
    return output_path
```

### 主流程 (main.py)

```python
# src/main.py
import os
import sys
from step1_download import download_video
from step2_transcribe import transcribe_audio
from step3_translate import translate_sentences
from step4_synthesize import synthesize_speech
from step5_merge import generate_srt, merge_video


def process_video(youtube_url: str, output_name: str = "output_video"):
    """完整处理流程"""

    print("=" * 50)
    print("🚀 YouTube 视频中文化处理开始")
    print("=" * 50)

    temp_dir = f"temp/{output_name}"
    output_dir = "output"
    os.makedirs(temp_dir, exist_ok=True)
    os.makedirs(output_dir, exist_ok=True)

    try:
        # Step 1: 下载视频
        print("\n[Step 1/5] 正在下载视频...")
        video_path, audio_path = download_video(youtube_url, temp_dir)
        print(f"✅ 视频下载完成: {video_path}")

        # Step 2: 语音识别
        print("\n[Step 2/5] 正在进行语音识别...")
        transcription = transcribe_audio(audio_path)
        sentences = transcription["sentences"]
        print(f"✅ 语音识别完成，共 {len(sentences)} 句")

        # Step 3: 翻译
        print("\n[Step 3/5] 正在进行翻译...")
        translated_sentences = translate_sentences(sentences)
        print(f"✅ 翻译完成")

        # 保存翻译结果 (可选)
        with open(f"{temp_dir}/translation.json", "w", encoding="utf-8") as f:
            import json
            json.dump(translated_sentences, f, ensure_ascii=False, indent=2)

        # Step 4: 语音合成
        print("\n[Step 4/5] 正在进行语音合成...")
        chinese_audio_path, audio_segments = synthesize_speech(
            translated_sentences,
            temp_dir
        )
        print(f"✅ 语音合成完成: {chinese_audio_path}")

        # Step 5: 生成字幕
        print("\n[Step 5/5] 正在生成字幕...")
        srt_path = generate_srt(translated_sentences, f"{temp_dir}/subtitle.srt")
        print(f"✅ 字幕生成完成: {srt_path}")

        # Step 6: 合成视频
        print("\n[Step 6/6] 正在合成最终视频...")
        final_output = f"{output_dir}/{output_name}.mp4"
        merge_video(video_path, chinese_audio_path, srt_path, final_output)

        print("\n" + "=" * 50)
        print("✅ 处理完成!")
        print(f"📁 输出文件: {final_output}")
        print("=" * 50)

        return final_output

    except Exception as e:
        print(f"\n❌ 处理失败: {e}")
        import traceback
        traceback.print_exc()
        raise


if __name__ == "__main__":
    # 示例用法
    if len(sys.argv) > 1:
        youtube_url = sys.argv[1]
    else:
        youtube_url = input("请输入 YouTube 视频 URL: ")

    output_name = "chinese_video"
    process_video(youtube_url, output_name)
```
